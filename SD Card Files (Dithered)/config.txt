[dither_palette]
# Chromacity weight mode (how it handles monochromatic vs highly chromatic palettes)
#  - dynamic is a dynamically weighted middle ground
#    > these middle grounds are good for making a particular color pop, but not necessarily
#      good for images with many vibrant colors. see also dynamic weight method.
#  - weighted is ideal for monochromatic palettes
#  - disabled is ideal for highly chromatic palettes with many colors
# dynamic|factorial|weighted|disabled
chromacity_mode = dynamic

# Dynamic weight method (how it calculates dynamic weights if chromacity mode is set to dynamic)
#  - Method 0 is best for somewhat monochrome palettes with few colors
#  - Method 1 is better than 0 for broad palettes, but still best with few colors
#  - Method 2 is similar to method 0
dynamic_weight_method = 1

dithering_enabled = true

# maps a palette's range of brightnesses to pure black and pure white.
# This will decrease contrast if enabled. doesn't work especially well.
map_brightness = false
